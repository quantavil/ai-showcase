def binary_search_first(arr, target):
    """
    Finds the first occurrence of target in sorted array arr.
    """
    low, high = 0, len(arr) - 1
    result = -1

    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            result = mid
            high = mid - 1 # Keep looking left
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
            
    return result
