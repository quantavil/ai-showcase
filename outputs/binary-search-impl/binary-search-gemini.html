def find_first_occurrence(data, val):
    if not data:
        return -1
    
    l, r = 0, len(data) - 1
    idx = -1
    
    while l <= r:
        m = l + (r - l) // 2
        if data[m] == val:
            idx = m
            r = m - 1
        elif data[m] < val:
            l = m + 1
        else:
            r = m - 1
            
    return idx
Time Cost O(log N)
